#include <stdio.h>
#include <conio.h>
#include <stdlib.h>
#include <string>	
#define SIZE 100

void NhapPhanTu(int &n)
{
	printf("Nhap Phan Tu:\n");
	scanf("%d", &n);
}

void NhapMang(int a[SIZE], int n)
{
	for (int i = 0; i < n; i++)
	{
		a[i] = rand() % 20 + 1;
	}
		
}

	
void XuatMang(int a[SIZE], int n)
{
	printf("Mang 1 chieu:\n");
	for (int i = 0; i < n; i++)
	{
		printf("%d\t", a[i]);
	}
}

void XoaDau(int a[SIZE], int &n)
{
	for (int i = 0; i <n-1; i++)
	{
		a[i] = a[i+1];
	}
	n--;
}

void XoaCuoi(int a[SIZE], int &n)
{
	n--;
}

void ThemDau(int a[SIZE], int &n, int x)
{
	for (int i = n; i > 0; i--)
	{
		a[i] = a[i - 1];
	}
	n++;
	a[0] = x;

}

void ThemCuoi(int a[SIZE], int &n, int x)
{
	a[n] = x;
	n++;
}

void main()
{
	int a[SIZE], n = 0;
	NhapPhanTu(n);
	NhapMang(a, n);
	XuatMang(a, n);
	XoaDau(a, n);
	XuatMang(a, n);
	XoaCuoi(a, n);
	XuatMang(a, n);
	ThemDau(a, n, 2);
	XuatMang(a, n);
	ThemCuoi(a, n, 8);
	XuatMang(a, n);
	_getch();
}
